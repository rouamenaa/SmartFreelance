pipeline {
    agent any

    environment {
        SONARQUBE_ENV = 'sonar'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'nassim-back', url: 'https://github.com/rouamenaa/Esprit-PIDEV-4SAE8-2026-SmartFreelance.git'
            }
        }

        stage('Build') {
            steps {
                dir('smartfreelance-backend/freelancer-service') {
                    sh 'mvn clean compile'
                }
            }
        }

        stage('Test') {
            steps {
                dir('smartfreelance-backend/freelancer-service') {
                    sh '''
                    mvn test \
                    -Dspring.profiles.active=test \
                    -Deureka.client.enabled=false \
                    -Dspring.cloud.discovery.enabled=false
                    '''
                }
            }
        }

        stage('SonarQube Analysis') {
            steps {
                dir('smartfreelance-backend/freelancer-service') {
                    withSonarQubeEnv("${SONARQUBE_ENV}") {
                        sh '''
                        mvn sonar:sonar \
                        -Dspring.profiles.active=test \
                        -Deureka.client.enabled=false \
                        -Dspring.cloud.discovery.enabled=false
                        '''
                    }
                }
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed!'
        }
    }
}
